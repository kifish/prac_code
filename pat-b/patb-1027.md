```c
#include <iostream>
#include <iomanip>
#include <string>
#include <math.h>
using namespace std;

int main(){
	int num;
	char a;
	cin >> num >> a;
	int n = sqrt((num+1)/2);
	for (int i = n; i >= 1; i--){
		for (int j = 0; j < n-i; j++)
			cout << " ";
		for (int j = 0; j < 2*i-1; j++)
			cout << a;
		cout << endl;
	}
	for (int i = 2; i <= n; i++){
		for (int j = 0; j < n-i; j++)
			cout << " ";
		for (int j = 0; j < 2*i-1; j++)
			cout << a;
		cout << endl;
	}
	cout << num-(2*n*n-1) << endl;
	return 0;
}
```
