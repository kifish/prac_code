```c
#include <iostream>
#include <iomanip>
#include <string.h>
using namespace std;

int main(){
	int n;
	int max_score = -1;
	int max_num = -1;
	cin >> n;
	int num[100010];
	memset(num, 0, sizeof(num));//不要写循环赋值，这样更快
	for (int i = 1; i <= n; i++){
		int temp, score;
		cin >> temp >> score;
		num[temp] += score;
		//cout << num[temp] << endl;
	}
	for (int i = 1; i <= n; i++){
		if (num[i] > max_score){
			max_score = num[i];
			max_num = i;
		}
	}
	cout << max_num << " " << max_score << endl;
	return 0;
}
```
