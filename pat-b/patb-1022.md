```c
#include <iostream>
#include <iomanip>
#include <cmath>
#include <cstring>
using namespace std;

int main(){
	int a, b, n, sum, res[100010], num;
	num = 0;
	cin >> a >> b >> n;
	sum = a + b;
	if (sum == 0)
		cout << "0";
	else{
		while(sum != 0){
			res[num] = sum%n;
			num++;
			sum = sum/n;
		}
		for (int i = num-1; i >= 0; i--)
			cout << res[i];
	}
	cout << endl;
	return 0;
}
```
