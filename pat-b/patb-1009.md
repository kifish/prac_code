
格式错误,

错误原因，这里如果遇到 EOF，那么idx还是会加1，words[idx-1]则存的是空字符串。

```C
#include <cstdio>
#include <string>
#include <iostream>
#include <cstring>
using namespace std;

char words[90][90];
int main(){
    //pat 是单点测试
    int idx = 0;
    while(scanf("%s",words[idx++]) != EOF);
    for(int i = idx -1;i>=0;i--){
        if(i != idx -1)
            printf(" %s",words[i]);
        else
            printf("%s",words[i]);
    }
//    printf("\n");
    return 0;
}
```
