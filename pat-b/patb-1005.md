'''
#include <iostream>
#include <iomanip>
using namespace std;

int main(){
	int sum;
	cin >> sum;
	int n[110], vis[100100], res[110];
	for (int i=0; i<sum; i++)
		cin >> n[i];
	for (int i=0; i<100100; i++)
		vis[i] = 0;
	for (int i=0; i<sum; i++){
		if(vis[n[i]] == 1)
			continue;
		int temp = n[i];
		while(temp != 1){
			if(temp%2==0){
				temp = temp/2;
				vis[temp] = 1;
			}
			else{
				temp = (temp*3+1)/2;
				vis[temp] = 1;
			}
		}
	}
	int num = 0;
	for (int i=0; i<sum; i++){
		if(vis[n[i]]==0){
			res[num] = n[i];
			num++;
		}
	}
	for (int i=0; i<num; i++){
		for (int j=0; j<i; j++){
			if (res[j]<res[i]){
				int temp = res[j];
				res[j] = res[i];
				res[i] = temp;
			}
		}
	}
	int flag = 0;
	for (int i=0; i<num; i++){
		if (flag == 0){
			cout << res[i];
			flag = 1;
		}
		else
			cout << " " << res[i];
	}
	cout << endl;
	return 0;
}
