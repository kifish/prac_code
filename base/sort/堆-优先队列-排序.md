class max_priority_q():
    def __init__(self):
        self.n = 0
        self.a = [-1]

    def swap(self, i, j):
        self.a[i], self.a[j] = self.a[j], self.a[i]

    def is_empty(self):
        return self.n < 1
    def swim(self,idx):
        """上浮"""
        while idx > 1 and self.a[idx] > self.a[idx//2]:
            self.swap(idx,idx//2)
            idx = idx // 2 
    def insert(self,val):
        self.a.append(val)
        self.n += 1
        self.swim(self.n)
    def sink(self,idx):
        """下沉"""
        while 2*idx <= self.n:
            max_child_idx = 2*idx 
            if max_child_idx < self.n and self.a[max_child_idx] < self.a[max_child_idx+1]:
                max_child_idx += 1

            if self.a[max_child_idx] > self.a[idx]:
                self.swap(max_child_idx,idx)
                idx = max_child_idx
            else:
                break
    def del_max(self):
        if self.n < 1:
            raise Exception
        if self.n == 1:
            self.n -= 1
            return self.a.pop()
        max_val = self.a[1]
        self.a[1] = self.a.pop() #若只有一个元素会出问题
        self.n -= 1
        self.sink(1)
        return max_val



if __name__ == '__main__':
    n = input()

    q = max_priority_q()
    for item in x:
        q.insert(item)
    res = []
    while not q.is_empty():
        res.append(q.del_max())
    res.reverse()
    s = ''
