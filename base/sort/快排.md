import random

def partition(a,lo,hi):
    i = lo
    j = hi + 1
    while True:
        i += 1
        while a[i] < a[lo] and i < hi:
            i += 1
        
        j -= 1
        while a[j] > a[lo] and j > lo:
            j -= 1 
        
        if i >= j:
            break
        a[i],a[j] = a[j],a[i]
    
    a[lo],a[j] = a[j],a[lo]
    return j 

def sort(a,lo,hi):
    if hi <= lo:
        return 
    j = partition(a,lo,hi)
    sort(a,lo,j-1)
    sort(a,j+1,hi)
    
def quicksort(a):
    random.shuffle(a) #保证pivot随机
    sort(a,0,len(a)-1)
    return a

import random
def quicksort_3_way(a):
    random.shuffle(a)
    sort(a,0,len(a)-1)
    return a 


"""
循环不变量
lt:当前已知的小于等于pivot的元素中的索引最小的索引
gt:当前已知的大于等于pivot的元素中的索引最小的索引
"""
def sort(a,lo,hi):
    if hi <= lo:
        return 
    i,lt,gt = lo,lo,hi 
    pivot = a[lo]
    while i <= gt:
        if a[i] < pivot:
            a[lt],a[i] = a[i],a[lt]
            lt += 1
            i += 1
        elif a[i] > pivot:
            a[i],a[gt] = a[gt],a[i]
            gt -= 1 
        else:
            i += 1
    sort(a,lo,lt-1)
    sort(a,gt+1,hi)


if __name__ == '__main__':
    test = [3, 6, 8, 10, 1, 2, 1]
    print(quicksort_3_way(test))
