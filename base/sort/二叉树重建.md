
由中根序列和后根序列重建二叉树
http: //algorithm.openjudge.cn/201801/E/
```c++

#include <cstdio>
#include <string>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#include <stack>
using namespace std;
vector<int> list1,list2;
stack<int> s;


string line;
string num_str;


struct Node{
    Node *left;
    Node *right;
    int key;
}node[1005];


int node_cnt;
Node* new_node(){
    node[node_cnt].left = NULL;
    node[node_cnt].right = NULL;
    return &node[node_cnt++];
}
//s1,e1 中序序列
//s2,e2 后序序列
Node* bulid_tree(int s1,int e1,int s2,int e2){
    Node* root = new_node();
    int root_idx = 0;
    for(;root_idx<=e1;root_idx++){
        if(list1[root_idx] == list2[e2])
            break;
    }
    root->key = list1[root_idx];
    //s2 +(root_idx  -s1)  s2 为左子树第一个节点的idx, root_idx - s1 为左子树节点的个数
    if(root_idx != s1)
        root->left = bulid_tree(s1,root_idx-1,s2,s2 +(root_idx - s1) -1); //s1,...,root_idx-1 即左子树的全部的节点
    if(root_idx != e1)
        root->right = bulid_tree(root_idx + 1,e1,s2+ root_idx-s1,e2-1);
    return root;
}


vector<int> res;
void preorder(Node* root){
    res.push_back(root->key);
    if(root->left)
        preorder(root->left);
    if(root->right)
        preorder(root->right);
}
int main(){
    node_cnt = 0;
    getline(cin,line);
    num_str.clear();
    for(int i =0;i<line.size();i++){
        if(line[i] != ' '){
            num_str.push_back(line[i]);
        }
        else{
            list1.push_back(stoi(num_str));
            num_str.clear();
        }
    }
    list1.push_back(stoi(num_str));

    num_str.clear();
    getline(cin,line);
    for(int i =0;i<line.size();i++){
        if(line[i] != ' '){
            num_str.push_back(line[i]);
        }
        else{
            list2.push_back(stoi(num_str));
            num_str.clear();
        }
    }
    list2.push_back(stoi(num_str));
    Node* root = bulid_tree(0,list1.size()-1,0,list2.size()-1);
    preorder(root);
    for(int i =0;i<res.size();i++){
        if(i != 0)
            cout<<" ";
        cout<<res[i];
    }
    return 0;
}
```
