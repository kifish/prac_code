poj2677测试数据有问题，多点测试是无法通过的，必须用单点测试。

```c
#include <cstdio>
int maze[105][105];
int main()
{
    int n;
    int S, L;
    scanf("%d", &n);
    //	while(scanf("%d",&n)!=EOF){
    S = 0;
    L = 0;
    for (int i = 0; i <= n + 1; i++)
    {
        for (int j = 0; j <= n + 1; j++)
        {
            maze[i][j] = 1;
        }
    }
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            int temp;
            scanf("%d", &temp);
            if (temp <= 50)
            {
                maze[i][j] = 0;
                S++;
            }
            else
            {
                maze[i][j] = 1;
            }
        }
    }
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            if (maze[i][j] == 0)
            {
                if (i == 1 || i == n || j == 1 || j == n)
                    L++;
                else if (maze[i - 1][j] + maze[i + 1][j] + maze[i][j - 1] + maze[i][j + 1])
                    L++;
            }
        }
    }
    printf("%d %d\n", S, L);
    //	}
    return 0;
}
```
