理解的时候要注意dp[i][j]的意思，i和j下标的意义。

实现的时候要注意字符串的index是从0开始的，因此可以考虑加个padding

```C
#include <iostream>
#include <iomanip>
#include <cmath>
#include <cstdio>
#include <queue>
#include <cstring>
#include <set>
#include <algorithm>
using namespace std;

int dp[201][201];
//dp[i][j] 表示string a前i个和string b 前j个字符可组成的最长公共子序列的长度。这里的i，j是数学意义上的。如果i为0，表示string a中不挑选字符。

int main() {
    string a;
    string b;
    while(cin>>a>>b){
        int len1 = a.size();
        int len2 = b.size();
        a = " " + a;
        b = " " + b;//padding
        for(int i = 0;i<=len1;i++){
            dp[i][0] = 0;
        }
        for(int j = 0;j<=len2;j++){
            dp[0][j] = 0;
        }

        for(int i = 1;i<=len1;i++){
            for(int j = 1;j<=len2;j++){
                if(a[i] == b[j]){
                    dp[i][j] = dp[i-1][j-1] + 1;
                }
                else{
                    dp[i][j] = max(dp[i-1][j],dp[i][j-1]);
                }
            }
        }
//        for(int i = 0;i<=len1;i++){
//            if(i == 0){

//            for(int j = 0;j<=len2;j++){
//                if(j == 0){
//                    if(i != 0) {
//                        cout<<a[i]<<" ";
//                    }
//                    else cout<<"  ";
//                }
//                cout<<dp[i][j]<<" ";
//            }
//            cout<<endl;
//        }
//        cout<<a<<endl<<b<<endl;
        cout<<dp[len1][len2]<<endl;
    }
    return 0;
}
```
