```C
#pragma warning(disable: 4996)
#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<map>
#include<queue>
#include<stack>
#include<vector>
#include<algorithm>
using namespace std;


struct Node {
	int x;
	int y;
};

int dx[] = { 0,0,-1,1 };
int dy[] = { 1,-1,0,0 };

char char_map[35][35];
int n;


bool inq[35][35] = { false };

bool is_valid1(int x, int y) {
	if (inq[x][y])
		return false;

	if (x > n || x < 1 || y > n || y < 1)
		return false;
	
	if (char_map[x][y] == '#')
		return false;
	else if (char_map[x][y] != 'r')
		return false;
	return true;
}
void bfs1(int x, int y) {
	Node tmp;
	tmp.x = x;
	tmp.y = y;
	queue<Node> q;
	q.push(tmp);
	inq[x][y] = 1;
	while (!q.empty()) {
		Node cur = q.front();
		q.pop();
		for (int i = 0; i < 4; i++) {
			int new_x = cur.x + dx[i];
			int new_y = cur.y + dy[i];
			if (is_valid1(new_x, new_y)) {
				tmp.x = new_x;
				tmp.y = new_y;
				inq[new_x][new_y] = 1;
				q.push(tmp);
			}
		}
	}
}


int bfs_travel1() {
	memset(inq, false, sizeof(inq));
	int cnt = 0;
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= n; j++) {
			if (is_valid1(i,j)) {
				bfs1(i, j);
				cnt++;
			}
		}
	}
	return cnt;
}



bool is_valid2(int x, int y) {
```

扩号匹配问题
```C
#pragma warning(disable: 4996)
#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<map>
#include<queue>
#include<stack>
#include<vector>
#include<algorithm>
using namespace std;

string raw;
char res[105];
string res_s;

void cal() {
	for (int i = 0; i < raw.size(); i++) {
		res[i] = ' ';
	}

	stack<int> s;
	for (int i = 0; i < raw.size(); i++) {
		if (raw[i] == '(') {
			s.push(i);
		}
		else if (raw[i] == ')') {
			if (s.size() >= 1) {
				s.pop();
			}
			else {
				res[i]= '?';
			}
		}
	}
	while (!s.empty()) {
		int idx = s.top();
		s.pop();
		res[idx] = '$';
	}
	res_s = res;
}
int main() {
	while (getline(cin,raw) && raw.size() >= 1)
	{
		cout << raw << endl;
		cal();
		cout<< res_s << endl;
	}
	//while (1);
	return 0;
}
```

```C
#pragma warning(disable: 4996)
#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<map>
#include<queue>
#include<stack>
#include<vector>
#include<algorithm>
#include<functional>
using namespace std;

int dp[305][35];


int min_cost[305][305];//中点建设邮局


int abs_val(int x) {
	if (x >= 0)
		return x;
	else  return -1 * x;
}
int loc[305];

const int INF = 0x3fffffff;
int main() {
	int n;
	cin >> n;
	int m;
	cin >> m;

	for (int i = 1; i <= n; i++) cin >> loc[i];

	for (int i = 1; i <= n; i++) {
		min_cost[i][i] = 0;
		for (int j = i + 1; j <= n; j++) {
			int mid_idx = (i + j) / 2;
			int sum = 0;
			for (int k = i; k <= j; k++) {
				sum += abs_val(loc[k] - loc[mid_idx]);
			}
			min_cost[i][j] = sum;
		}
	}
	for (int i = 1; i <= n; i++) {
		dp[i][0] = INF;
		dp[i][1] = min_cost[1][i];
	}
	for (int i = 1; i <= n; i++) {
		for (int j = 2; j <= m; j++) {
			int min_val = dp[i][j - 1];
			for (int k = 1; k + 1 <= i; k++) {
				min_val = min(min_val, dp[k][j-1] + min_cost[k+1][i]);
				//cout << "i " << i << " j " << j << " k " << k << endl;;
				
			}
			//cout << "----------" << endl;
			//cout << dp[i][j] << endl;
			dp[i][j] = min_val;
		}
	}
	cout << dp[n][m] << endl;
	//while (1);
	return 0;
}
```
