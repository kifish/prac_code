```c
   memset(vis,0, sizeof(vis));
}
int delete_idx;
void dfs(int idx){
    vis[idx] = 1;
    for(int i = 0;i<adj[idx].size();i++){
        if(!vis[adj[idx][i]] && delete_idx != adj[idx][i]){
            dfs(adj[idx][i]);
        }
    }
}

int dfs_travel(){
    int cnt = 0;
    for(int i = 1;i<=n;i++){
        if(!vis[i] && delete_idx != i){
            cnt++;
            dfs(i);
        }
    }
    return cnt;
}
int main(){
    scanf("%d %d %d",&n,&m,&k);
    int idx1,idx2;
    while (m--){
        scanf("%d %d",&idx1,&idx2);
        adj[idx1].push_back(idx2);
        adj[idx2].push_back(idx1);
    }
    while(k--){
        scanf("%d",&delete_idx);
        init();
        printf("%d\n",dfs_travel()-1);
    }
    return 0;
}
```

```C
#include <iostream>
#include <iomanip>
#include <cmath>
#include <cstdio>
#include <queue>
#include <cstring>
#include <string>
#include <vector>
#include <algorithm>
#include <set>
#include <map>
using namespace std;

int n,m,k;

vector<int> adj[1005];
bool is_recorded[1005] = {false};
int parent[1005];


void init(){
   memset(is_recorded,0, sizeof(is_recorded));
   for(int i = 1;i<=n;i++) parent[i] = i;
}
int delete_idx;

int find_root(int x){
    if(parent[x] == x) return x;
    int root = find_root(parent[x]);
    parent[x] = root;
    return root;
}

void build(){
    for(int i = 1;i<=n;i++){
        if(i == delete_idx) continue;
        for(int j = 0;j<adj[i].size();j++){
            if(adj[i][j] == delete_idx) continue;
            if(find_root(i) != find_root(adj[i][j])){
                parent[i] = adj[i][j];
            }
        }
    }
}



//测试用例:

int set_cnts(){
    int cnt = 0;
    int left_city = 0;
    for(int i = 1;i<=n;i++){
        if(i == delete_idx) continue;
        left_city++;
        if(is_recorded[find_root(i)] == false){
            cnt++;
            is_recorded[find_root(i)] = 1;
        }
    }
    if(left_city)
        return cnt;
    return 1;
}

int main(){
    scanf("%d %d %d",&n,&m,&k);
    int idx1,idx2;
    while (m--){
        scanf("%d %d",&idx1,&idx2);
        adj[idx1].push_back(idx2);
        adj[idx2].push_back(idx1);//必须要变成无向图。这行不加会出问题。
    }
    while(k--){
        scanf("%d",&delete_idx);
        init();
        build();
        printf("%d\n",set_cnts()-1);
    }
    return 0;
}
```
