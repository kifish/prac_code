#include <iostream>
#include <iomanip>
#include <cmath>
#include <cstdio>
#include <queue>
#include <cstring>
#include <string>
#include <vector>
#include <algorithm>
#include <set>
#include <map>
using namespace std;

struct Stu{
    int id;
    int s1;
    int s2;
    int sum; // /2
    int choice[6];
}stu[40005];

int school[105];
int pre_rank[105] = {};

bool cmp(Stu &a,Stu &b){
    if(a.sum != b.sum)
        return a.sum > b.sum;
    else
        return a.s1 > b.s1;
}
int main(){
    int n,k,m;
    scanf("%d %d %d",&n,&m,&k);
    vector<int> res[m];
    for(int i = 0;i<m;i++) scanf("%d",&school[i]);
    for(int i = 0;i<n;i++) {
        scanf("%d %d",&stu[i].s1,&stu[i].s2);
        stu[i].sum = stu[i].s1 + stu[i].s2;
        stu[i].id = i;
        for(int j = 0;j< k;j++) scanf("%d",&stu[i].choice[j]);
    }
    sort(stu,stu+n,cmp);
    int rank = 1;
    for(int i = 0;i<n;i++){
        if(!(i == 0 || stu[i].sum == stu[i-1].sum && stu[i].s1 == stu[i-1].s1))
            rank = i + 1;//更新rank
        for(int j = 0;j<k;j++){
            if(school[stu[i].choice[j]] > 0 || rank == pre_rank[stu[i].choice[j]]){
                res[stu[i].choice[j]].push_back(stu[i].id);
                pre_rank[stu[i].choice[j]] = rank;
                school[stu[i].choice[j]]--;
                break;
            }
        }
    }
    for(int i = 0;i<m;i++){
        if(res[i].size() == 0)
            printf("\n");
        else{
            sort(res[i].begin(),res[i].end());
            for(int j = 0;j<res[i].size();j++){
                if(j != 0) printf(" ");
                printf("%d",res[i][j]);
            }
            printf("\n");
        }
    }
    return 0;
}
