#include <iostream>

using namespace std;

const int INF = 0x3fffffff;
const int MAXV = 1015;


struct Node{
    int v;
    int w;
};


vector<Node> adj[MAXV];
bool got[MAXV];
int d[MAXV];
int n;

void dijkstra(int s){
    fill(d,d+MAXV,INF);
    memset(got, false,sizeof(got));
    d[s] = 0;
    int times = n;
    while (times--){
        int u = -1,min_d = INF;
        for(int i = 1;i<=n;i++){
            if(!got[i] && d[i] < min_d){
                min_d = d[i];
                u = i;
            }
        }
        if(u == -1) return;
        got[u] = 1;
        for(int i = 0;i<adj[u].size();i++){
            int v = adj[u][i].v;
            int w = adj[u][i].w;
            if(!got[v]){
                if(d[u] + w < d[v]){
                    d[v] = d[u] + w;
                }
                else if(d[u] + w == d[v]){
                    ;
                }
            }
        }
    }
}








int main(){
    int n_house;
    int n_location;
    int m;
    int service_range;
    scanf("%d %d %d %d",&n_house,&n_location,&m,&service_range);
    n = n_house + n_location;
    int u,v,edge_weight;
    char str1[6];
    char str2[6];
    string s1,s2;
    while (m--){
        scanf("%s %s %d",str1,str2,&edge_weight);
        s1 = str1,s2 = str2;
        if(s1[0] == 'G') u = stoi(s1.substr(1,s1.size())) + n_house;
        else u = stoi(s1);
        if(s2[0] == 'G') v = stoi(s2.substr(1,s2.size())) + n_house;
        else v = stoi(s2);
        Node tmp;
        tmp.w = edge_weight;
        tmp.v = v;
        adj[u].push_back(tmp);
        tmp.v = u;
        adj[v].push_back(tmp);
    }
    int g_max_min_dis;
    double g_min_avg_dis;
    int opt_loc_idx = -1;//-1 : no solution
    for(int i = n_house + 1;i<=n;i++){
        int s = i;
        dijkstra(s);
        bool valid = true;
        int min_d = INF;
        double avg_dis = 0;
//        int sum = 0;
        for(int j = 1;j<=n_house;j++){
            if(d[j] > service_range) {
                valid = false;
                break;
            }
            if(d[j] < min_d) min_d = d[j];
//            sum += d[j];
            avg_dis += d[j];
        }
        if(!valid) continue;

        avg_dis /= n_house;
//        avg_dis = 1.0 * sum / n_house;
        if(opt_loc_idx == -1){
            g_max_min_dis = min_d;
            g_min_avg_dis = avg_dis;
            opt_loc_idx = i;
            continue;
        }
        if(min_d > g_max_min_dis){
            g_max_min_dis = min_d;
            g_min_avg_dis = avg_dis;
            opt_loc_idx = i;
        }
        else if(min_d == g_max_min_dis && avg_dis < g_min_avg_dis){
            g_max_min_dis = min_d;
            g_min_avg_dis = avg_dis;
            opt_loc_idx = i;
        }
    }
    if(opt_loc_idx == -1){
        printf("No Solution\n");
    }
    else{
        printf("G%d\n",opt_loc_idx - n_house);
        printf("%d.0 %.1f\n",g_max_min_dis,g_min_avg_dis);
    }
    return 0;
}
