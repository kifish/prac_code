```c
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

int main(){
	int num, score;
	int max_score = -1;
	int min_score = 100;
	string max_name, max_id, min_name, min_id, name, gender, id;
	int find_max = 0;
	int find_min = 0;
	cin >> num;
	while(num--){
		cin >> name >> gender >> id >> score;
		if (gender == "F"){
			find_max = 1;
			if (score > max_score){
				max_score = score;
				max_name = name;
				max_id = id;
			}
		}
		if (gender == "M"){
			find_min = 1;
			if (score < min_score){
				min_score = score;
				min_name = name;
				min_id = id;
			}
		}
	}
	if (find_max == 1)
		cout << max_name << " " << max_id << endl;
	else
		cout << "Absent" << endl;
	if (find_min == 1)
		cout << min_name << " " << min_id << endl;
	else
		cout << "Absent" << endl;
	if (find_min == 1 && find_max == 1)
		cout << max_score-min_score << endl;
	else
		cout << "NA" << endl;
	return 0;
}
```
