#include <iostream>
#include <iomanip>
#include <cmath>
#include <queue>
#include <cstring>
using namespace std;

int k, p;
int max_num = -1;
vector<int> res, temp;

void dps(int n, int num){
	//cout << n << " " << num  << " " << temp.size() << endl;
	if (n == 0 && temp.size() == k){
		int sum = 0;
		for (int i = 0; i < temp.size(); i++)
			sum += temp[i];
		if (sum > max_num){
			res = temp;
			max_num = sum;
		}
		return;
	}
	if (n <= 0 || temp.size() >= k || num <= 0) return;
    temp.push_back(num);
    dps(n-pow(num, p), num);
    temp.pop_back();
    dps(n, num-1);
    return;
}

int main(){
    int n;
    cin >> n >> k >> p;
    int num = pow(n, 1.0/float(p));
    //cout << num << endl;
    dps(n, num);
    if (res.size() != 0){
	    cout << n << " = " << res[0] << "^" << p;
	    for (int i = 1; i < res.size(); i++)
	    	cout << " + " << res[i] << "^" << p;
	}
	else
		cout << "Impossible";
    cout << endl; 
    return 0;
}
