```c
#include <iostream>
#include <iomanip>
#include <string>
#include <math.h>
using namespace std;

int main(){
	int num;
	string s;
	cin >> s;
	int len = s.length();
	num = (len-1)/3;
	int n = len - num*2 - 2;
	for (int i = 0; i < num; i++){
		cout << s[i];
		for (int j = 0; j < n; j++)
			cout << " ";
		cout << s[len-1-i] << endl;
	}
	for (int i = 0; i < n+2; i++){
		cout << s[num+i];
	}
	cout << endl;
	return 0;
}
```
