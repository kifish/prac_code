```c
#include <iostream>
#include <iomanip>
#include <queue>
#include <cstring>
using namespace std;

struct Node{
    int h;
    int a;
    int b;
};

int m, n, l, t;
int sum = 0;
int num[70][1300][140], vis[70][1300][140];

int main(){
    cin >> m >> n >> l >> t;
    //memset(vis, 0, sizeof(vis));
    for (int i = 0; i < l; i++)
        for (int j = 0; j < m; j++)
            for (int k = 0; k < n; k++) {
                cin >> num[i][j][k];
                vis[i][j][k] = 0;
            }
    for (int i = 0; i < l; i++) {
        for (int j = 0; j < m; j++) {
            for (int k = 0; k < n; k++) {
                if (vis[i][j][k] == 1 || num[i][j][k] == 0)
                    continue;
                Node node;
                queue<Node> q;
                node.h = i;
                node.a = j;
                node.b = k;
                q.push(node);
                int count = 1;
                vis[i][j][k] = 1;
                while (!q.empty()) {
                    Node temp = q.front();
                    int height = temp.h;
                    int x = temp.a;
                    int y = temp.b;
                    q.pop();
                    if (height - 1 >= 0) {
                        if (vis[height - 1][x][y] == 0 && num[height - 1][x][y] == 1) {
                            count++;
                            Node neighbour;
                            neighbour.h = height - 1;
                            neighbour.a = x;
                            neighbour.b = y;
                            vis[neighbour.h][neighbour.a][neighbour.b] = 1;
                            q.push(neighbour);
                        }
                    }
                    if (height + 1 < l) {
                        if (vis[height + 1][x][y] == 0 && num[height + 1][x][y] == 1) {
                            count++;
                            Node neighbour;
                            neighbour.h = height + 1;
                            neighbour.a = x;
                            neighbour.b = y;
                            vis[neighbour.h][neighbour.a][neighbour.b] = 1;
                            q.push(neighbour);
                        }
                    }
                    if (x - 1 >= 0) {
                        if (vis[height][x - 1][y] == 0 && num[height][x - 1][y] == 1) {
                            count++;
                            Node neighbour;
                            neighbour.h = height;
                            neighbour.a = x - 1;
                            neighbour.b = y;
                            vis[neighbour.h][neighbour.a][neighbour.b] = 1;
                            q.push(neighbour);
                        }
                    }
                    if (x + 1 < m) {
                        if (vis[height][x + 1][y] == 0 && num[height][x + 1][y] == 1) {
                            count++;
                            Node neighbour;
                            neighbour.h = height;
                            neighbour.a = x + 1;
                            neighbour.b = y;
                            vis[neighbour.h][neighbour.a][neighbour.b] = 1;
                            q.push(neighbour);
                        }
                    }
                    if (y - 1 >= 0) {
                        if (vis[height][x][y - 1] == 0 && num[height][x][y - 1] == 1) {
                            count++;
                            Node neighbour;
                            neighbour.h = height;
                            neighbour.a = x;
                            neighbour.b = y - 1;
                            vis[neighbour.h][neighbour.a][neighbour.b] = 1;
                            q.push(neighbour);
                        }
                    }
                    if (y + 1 < n) {
                        if (vis[height][x][y + 1] == 0 && num[height][x][y + 1] == 1) {
                            count++;
                            Node neighbour;
                            neighbour.h = height;
                            neighbour.a = x;
                            neighbour.b = y + 1;
                            vis[neighbour.h][neighbour.a][neighbour.b] = 1;
                            q.push(neighbour);
                        }
                    }
                }
                if (count >= t)
                    sum += count;
            }
        }
    }
    cout << sum << endl;
    return 0;
}
```
